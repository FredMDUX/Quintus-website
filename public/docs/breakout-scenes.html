    Level Skipping

      Q.input.on('left',function() {
    var level = Q.state.get("level");

    if(level > 1) {
      Q.stageScene('level' + (level-1));
    } else {
      Q.stageScene('title');
    }
  });

  Q.input.on('right',function() {
    var level = Q.state.get("level") || 0;

    if(level < 4) {
      Q.stageScene('level' + (level+1));
    } else {
      Q.stageScene('winner');
    }
  });
